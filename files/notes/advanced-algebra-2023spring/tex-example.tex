\begin{Prob}[P3 第7题]在环$R$中, 证明
    \[0a=a0=0,\,\forall a\in R.\]
\end{Prob}
\begin{proof}
    由于
    \[0a=(0+0)a=0a+0a,\]
    那么
    \[0=0a+(-0a)=0a+0a+(-0a)=0a.\]
    同理可得$a0=0$, 故原命题成立.
\end{proof}
